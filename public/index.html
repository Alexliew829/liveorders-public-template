import { initializeApp, cert, getApps } from 'firebase-admin/app'; import { getFirestore } from 'firebase-admin/firestore'; import { format } from 'date-fns'; import { zhCN } from 'date-fns/locale'; import ExcelJS from 'exceljs'; const serviceAccount = JSON.parse(process.env.FIREBASE_ADMIN_KEY); if (!getApps().length) { initializeApp({ credential: cert(serviceAccount) }); } const db = getFirestore(); export default async function handler(req, res) { try { const snapshot = await db.collection('triggered_comments').get(); if (snapshot.empty) { return res.status(200).json({ message: '没有订单数据可导出！' }); } const workbook = new ExcelJS.Workbook(); const sheet = workbook.addWorksheet('订单'); // 设置列格式 sheet.columns = [ { header: '顾客名称', key: 'user_name', width: 20 }, { header: '商品编号', key: 'selling_id', width: 15 }, { header: '商品名称', key: 'product_name', width: 30 }, { header: '数量', key: 'quantity', width: 10 }, { header: '价格', key: 'price', width: 15 }, { header: '已发送连接', key: 'replied', width: 15 }, ]; snapshot.forEach((doc) => { const data = doc.data(); sheet.addRow({ user_name: data.user_name || '', selling_id: data.selling_id || '', product_name: data.product_name || '', quantity: 1, price: data.price || '', replied: data.replied ? '✅' : '❌', }); }); const buffer = await workbook.xlsx.writeBuffer(); const today = format(new Date(), 'dd-MM-yy', { locale: zhCN }); const filename = `${today} Bonsai-Order.xlsx`; res.setHeader('Content-Type', 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'); res.setHeader('Content-Disposition', `attachment; filename="${filename}"`); res.status(200).send(Buffer.from(buffer)); } catch (err) { console.error('导出失败:', err); res.status(500).json({ error: '导出失败', detail: err.message }); } }
